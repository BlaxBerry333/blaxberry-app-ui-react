### 基本使用

需结合`<BaseListItem>`组件一起使用

<Story of={Stories.ExampleBase} />

```tsx
import * as React from "react";
import { BaseList, BaseListItem } from "blaxberry-app-ui-react";

function Component() {
  return (
    <BaseList
      listProps={{
        style: {
          border: "1px solid grey",
          padding: 8,
          width: 200,
        },
      }}
      listSubheaderProps={{
        children: "副标题",
      }}
    >
      <BaseListItem primaryContent="上山打老虎" secondaryContent="老虎没打着" />

      <BaseListItem
        primaryContent={
          <>
            <strong>自定义内容</strong>
            <span>自定义内容</span>
            <small>自定义内容</small>
          </>
        }
      />

      <BaseListItem
        primaryContent="点击"
        listItemButtonProps={{
          onClick: (e) => {
            e.stopPropagation();
            console.log("onClick");
          },
        }}
      />

      <BaseListItem
        primaryContent="拖拽"
        listItemButtonProps={{
          disableRipple: true, // https://github.com/react-dnd/react-dnd/issues/832
          // disableTouchRipple: true,
          style: { cursor: "grab" },
          draggable: true,
          onDragStart: (e) => {
            e.stopPropagation();
            console.log("onDragStart");
          },
          onDragEnd: (e) => {
            e.stopPropagation();
            console.log("onDragEnd");
          },
        }}
      />
    </BaseList>
  );
}
```
