## Installation

```tsx
yarn add git+https://github.com/BlaxBerry333/blaxberry-app-ui-react
```

## Usage

```tsx
import {
  // components
  AdminLayout,
  BaseButton,
  // types
  type AdminLayoutProps,
  // static objects
  AdminAppStyle,
  // functions
  toggleFullScreen,
} from "blaxberry-app-ui-react";
```

## Development Flow

```shell
# 1. check
yarn check-prettier && yarn check-eslint
# 2. build lib
yarn build:lib
# 3. commit then push
git commit -m "commit message"
git push
```
